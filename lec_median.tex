\begin{frame}{Mediane und das Auswahlproblem}
	\begin{exampleblock}{Das Auswahlproblem}
		\begin{itemize}
			\item gegeben ist ein Array $\vA=(A_1,\ldots,A_n)$ und eine Zahl $1\leq\ell\leq n$
			\item die Elemente sind vergleichbar
			\item das Ziel ist, das $\ell$-te Element von $\vA$ in aufsteigender Reihenfolge zu finden
		\end{itemize}
	\end{exampleblock}
\end{frame}

\begin{frame}{Mediane und das Auswahlproblem}
	\begin{exampleblock}{L\"osung via Sortieren}
		\begin{itemize}
			\item offenbar k\"onnten wir $\vA$ in Zeit $O(n\log n)$ sortieren und einfach den $\ell$-ten Eintrag des sortierten Arrays ausgeben
			\item aber f\"ur $\ell=1$ und $\ell=n$ kann das Problem in Zeit $O(n)$ gel\"ost werden
			\item geht das auch f\"ur allgemeine $\ell$?
		\end{itemize}
	\end{exampleblock}
\end{frame}

\begin{frame}{Mediane und das Auswahlproblem}
	\begin{block}{Definition}
		Ein \alert{Median} eines Arrays $\vA=(A_1,\ldots,A_n)$ ist ein Element $A_i$ von $\vA$, so da\ss\
		\begin{align*}
			\abs{\cbc{i\in[n]:A_i<m}}\leq\frac n2\quad\mbox{und}\quad\abs{\cbc{i\in[n]:A_i>m}}\leq\frac n2.
		\end{align*}
	\end{block}
	\begin{exampleblock}{}
		\begin{itemize}
			\item wenn wir das Array sortieren, ist das Element in Position $\lceil(n+1)/2\rceil$ ein Median
			\item wenn $n$ ungerade ist und alle Elemente verschieden sind, ist der Median eindeutig
			\item wenn $n$ gerade ist, gibt es mindestens zwei Mediane
		\end{itemize}
	\end{exampleblock}
\end{frame}

\begin{frame}{Mediane und das Auswahlproblem}
	\begin{exampleblock}{{\tt Select}$(\vA,\ell)$}
	\begin{enumerate}
		\item Falls $n=1$, gib $A_1$ aus.
		\item Unterteile $\vA$ in $k=\lfloor n/5\rfloor$ Teilarrays $T_1,\ldots,T_{\lfloor n/5\rfloor}$ zu je $5$ Elementen und, falls $n$ nicht durch 5 teilbar ist, ein weiteres Teilarray $T_{\lfloor n/5\rfloor+1}$ auf.
		\item Setze $N=\lfloor n/5\rfloor$, falls $n$ durch 5 teilbar ist, und $N=\lfloor n/5\rfloor+1$ sonst.
		\item Finde in jedem dieser Teilarray $T_i$ einen Median $m_i$.
	\end{enumerate}
	\end{exampleblock}
\end{frame}

\begin{frame}{Mediane und das Auswahlproblem}
	\begin{exampleblock}{{\tt Select}$(\vA,\ell)$}
	\begin{enumerate}
		\setcounter{enumi}4
		\item Wende {\tt Select} rekursiv an, um einen Median $m$ von $\vm=(m_1,\ldots,m_N)$ zu finden.
		\item Bestimme $\vK=(K_1,\ldots,K_{n'}),\vM=(M_1,\ldots,M_{n''}),\vG=(G_1,\ldots,G_{n'''})$, die die Elemente von $\vA$ kleiner/gleich/gr\"o\ss er $m$ enthalten.
		\item Falls $n'\geq\ell$, f\"uhre {\tt Select}$(\vK,\ell)$ aus;
		\item sonst, falls $n'+n''\geq\ell$, gib $m$ aus;
		\item sonst f\"uhre {\tt Select}$(\vG,\ell-n'-n'')$ aus.
	\end{enumerate}
	\end{exampleblock}
\end{frame}

\begin{frame}{Mediane und das Auswahlproblem}
	\begin{block}{Satz}
	{\tt Select}$(\vA,\ell)$ hat Laufzeit $O(n)$.
	\end{block}
\end{frame}

\begin{frame}{Mediane und das Auswahlproblem}
	\begin{exampleblock}{Zusammenfassung}
		\begin{itemize}
			\item {\tt Select} bestimmt das $\ell$-te Element in linearer Zeit
			\item es gibt einen einfacheren randomisierten Algorithmus f\"ur dieses Problem
			\item dieser \"ahnelt Quicksort, aber ``verfolgt nur einen Zweig''
		\end{itemize}
	\end{exampleblock}
\end{frame}
