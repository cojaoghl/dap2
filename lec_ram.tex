\begin{frame}{Die Registermaschine}
	\begin{exampleblock}{Aufbau}
		\begin{itemize}
			\item Register $(r_i)_{i\geq0}$; jedes speichert eine ganze Zahl, initialisiert auf Null 
			\item $r_0$ ist der ``Akkumulator''
			\item ferner werden in Register $r_0$ die Ein- und Ausgabe placiert
			\item einen Programmz\"ahler $z$
			\item Programm aus Zeilen, die jeweils einen Befehl enthalten
		\end{itemize}
	\end{exampleblock}
\end{frame}

\begin{frame}{Die Registermaschine}
\begin{center}
\begin{tabular}{|l|l|}\hline
	{\tt read} $j$&schreibe den Wert von $r_j$ in $r_0$\\
	{\tt read} $*j$&wenn $h$ der Wert von $r_j$ ist, schreibe den Wert von $r_h$ in $r_0$\\
	{\tt store} $j$&schreibe den Wert von $r_0$ in $r_j$\\
	{\tt store} $*j$&wenn $h$ der Wert von $r_j$ ist, schreibe den Wert von $r_0$ in $r_h$\\
	{\tt load} $x$&schreibe die Zahl $x$ in $r_0$\\
	{\tt add} $x$&addiere $x$ zu der Zahl in $r_0$\\
	{\tt half}&wenn $y$ der Wert von $r_0$ ist, setze $r_0$ auf $\lfloor y/2\rfloor$\\
	{\tt jump} $j$&setze den Programmz\"ahler auf den Wert $j$\\
	{\tt jpos} $j$&wenn der Wert in $r_0$ positiv ist, f\"uhre {\tt jump} $j$ aus\\
	{\tt jneg} $j$&wenn der Wert in $r_0$ negativ ist, f\"uhre {\tt jump} $j$ aus\\
	{\tt jzero} $j$&wenn der Wert in $r_0$ gleich Null ist, f\"uhre {\tt jump} $j$ aus\\
	{\tt halt}&beende das Programm\\\hline
\end{tabular}
\end{center}
\end{frame}

\begin{frame}{Die Registermaschine}
	\begin{exampleblock}{Laufzeit}
		\begin{itemize}
			\item die \alert{Laufzeit} $T_{\cM}(e)$ einer Registermaschine $\cM$ auf Eingabe $e\in\ZZ$ ist die Zahl der ausgef\"uhrten Befehle
			\item also $T_{\cM}(e)\in\NN\cup\cbc\infty$
			\item die \alert{Eingabel\"ange} einer Zahl $e\in\ZZ$ ist $\lceil\log_2(2+|e|)\rceil$
			\item {\em Anzahl Bits, um $e$ hinzuschreiben}
		\end{itemize}
	\end{exampleblock}
\end{frame}

\begin{frame}{Die Registermaschine}
	\begin{exampleblock}{Effiziente Algorithmen}
		\begin{itemize}
			\item $\cT_{\cM}(n)$ ist die maximale Laufzeit von $\cM$ auf einer Eingabe der L\"ange $\leq n$:
				\begin{align*}
					\cT_{\cM}(n)&=\max\cbc{T_\cM(e):\log_2(2+|e|)\leq n}
				\end{align*}
			\item der ``Algorithmus'' $\cM$ ist \alert{effizient}, wenn es eine Zahl $\ell>0$ gibt, so da\ss\ 
				\begin{align*}
					\cT_{\cM}(n)=O(n^\ell)
				\end{align*}
		\end{itemize}
	\end{exampleblock}
\end{frame}

\begin{frame}{Die Registermaschine}
	\begin{block}{Die Klasse P}
		\begin{itemize}
			\item $f:\ZZ\to\ZZ$ hei\ss t \alert{effizient berechenbar}, wenn es einen effizienten Algorithmus $\cM$ gibt, der f\"ur alle $e\in\ZZ$ bei Eingabe $e$ den Wert $f(e)$ ausgibt
			\item P ist die Menge aller effizient berechenbaren Funktionen $f:\ZZ\to\{0,1\}$
		\end{itemize}
	\end{block}
\end{frame}
